[문제](https://www.acmicpc.net/problem/1780)

```c{3}
#include <cstdio>

const int MAX = 2187;

int answer[4] = { 0,0,0,0 };
int arr[MAX][MAX];
void recur(int size, int x, int y) {
	int cur = arr[x][y];
	bool flag = true;
	if (size == 1) {
			answer[cur+2] += 1;
		return;
	}
	for (int i = x; i < x + size; i++) {
		for (int j = y; j < y + size; j++) {
			if (arr[i][j] != cur) {
				flag = false;
				if (size == 1)
					return;
				for (int k = x; k < x+size; k += size / 3) {
					for (int l = y; l < y+size; l += size / 3) {
						recur(size / 3, k, l);
					}
				}
				return;
			}
		}
	}
	if (flag) {
		answer[cur + 2] += 1;
		}
	}

int main() {
	int N;
	scanf("%d", &N);
	for (int i = 0; i < N; i++) {
		for (int j = 0; j < N; j++) {
			scanf("%d", &arr[i][j]);
		}
	}
	recur(N, 0, 0);
	for (int i = 1; i < 4; i++)
		printf("%d\n", answer[i]);
	return 0;
}
```

- 재귀로 분할정복을 이용해 해결했습니다.
- N의 최댓값인 2187보다 작게 길이를 잡으면 시간초과가 납니다.(실수)
